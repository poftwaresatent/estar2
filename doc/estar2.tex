\begin{algorithm}
  \caption{Main}\label{algo:main}
  \begin{algorithmic}
    \LOOP
      \STATE ProcessChanges()
      \STATE Propagate()
      \IF { TopKey($Q$) $> v(r)$ }
        \STATE status $\leftarrow$ \textsc{Success}
      \ELSIF { Empty($Q$) }
        \STATE status $\leftarrow$ \textsc{Failure}
      \ELSE
        \STATE status $\leftarrow$ \textsc{Pending}
      \ENDIF
    \ENDLOOP
  \end{algorithmic}
\end{algorithm}

\begin{algorithm}
  \caption{Propagate}\label{algo:propagate}
  \begin{algorithmic}
    \IF { not Empty($Q$) }
      \STATE $x \leftarrow$ Extract($Q$)
      \IF { $v(x) > \rhs(x)$ }
        \STATE $v(x) \leftarrow \rhs(x)$
        \FORALL { $n \in N(x)$ }
          \STATE Update($n$)
        \ENDFOR
      \ELSE
        \STATE $v(x) \leftarrow \infty$
        \FORALL { $n \in N(x)$ }
          \STATE Update($n$)
        \ENDFOR
        \STATE Update($x$)
      \ENDIF
    \ENDIF
  \end{algorithmic}
\end{algorithm}

\begin{algorithm}
  \caption{Update($x$)}\label{algo:update}
  \begin{algorithmic}
    \IF { $x \in G \cup O$ }
      \STATE Remove($Q, x$)
%      \COMMENT { can be no-op }
    \ELSE
      \STATE $\rhs(x) \leftarrow \min_{\{a,b\} \in P(x)} T(v(a),v(b))$
      \IF { $v(x) \ne \rhs(x)$ }
        \STATE Insert($Q, x$)
%        \COMMENT { can be update }
      \ELSE
        \STATE Remove($Q, x$)
%        \COMMENT { can be no-op }
      \ENDIF
    \ENDIF
  \end{algorithmic}
\end{algorithm}

\begin{equation}
  T(A,B) =
  \begin{cases}
    A + h/F                                & \Leftarrow h/F \leq B-A \\
    \big(\beta + \sqrt{\beta^2 - 2\gamma}\big) / 2 & \text{otherwise}
  \end{cases}
\end{equation}
\noindent where
\begin{subequations}
  \begin{align}
    \beta &= A + B \\
    \gamma &= A^2 + B^2 - h^2/F^2
  \end{align}
\end{subequations}
